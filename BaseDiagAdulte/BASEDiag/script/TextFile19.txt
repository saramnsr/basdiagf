ALTER TABLE ENVOI_ENTENTE
ADD ID_MODELE_ENVOI INTEGER;


ALTER TABLE BASEDIAG_DIAGNOSTIC
ADD ID_MODELE_ENTENTE INTEGER;

update BASEDIAG_DIAGNOSTIC set ID_MODELE_ENTENTE = (select first 1 ID from ENVOI_ENTENTE where EE_DEBUTTRAITEMENT = 1 and EE_PATIENT=BASEDIAG_DIAGNOSTIC.ID_PATIENT)
where ID_MODELE_ENTENTE is null;

update ENVOI_ENTENTE set ID_MODELE_ENVOI = (select first 1 ID from MODELE_ENTENTE where MODELE_ENTENTE.me_patient =ENVOI_ENTENTE.EE_PATIENT order by MODELE_ENTENTE.id asc)
where ID_MODELE_ENVOI is null;
